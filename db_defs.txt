DROP DATABASE IF EXIST CDWDEF;
CREATE DATABASE CDWDEF;

CREATE TABLE CDWDEF.DBASES (
  id INT(4) NOT NULL PRIMARY KEY,
  dname VARCHAR(32)
);

CREATE TABLE CDWDEF.TABLES (
  id INT(4) NOT NULL PRIMARY KEY,
  tname VARCHAR(32),
  did INT(4),
  vcount INT(4)
);

CREATE TABLE CDWDEF.VARS (
  id INT(4) NOT NULL PRIMARY KEY,
  vname VARCHAR(32),
  vtype VARCHAR(32)
);

CREATE TABLE CDWDEF.TV (
  id INT(4) NOT NULL PRIMARY KEY,
  tid INT(4),
  vid INT(4)
);
